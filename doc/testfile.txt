$Error.Clear()

function Test-SingleInstance {
    param([string]$MutexName = "RUNNING")

    $mutex = New-Object System.Threading.Mutex($false, $MutexName)
    $acquired = $mutex.WaitOne(0)

    return @{
        Mutex = $mutex
        Acquired = $acquired
    }
}
