$Error.Clear()

function Test-SingleInstance {
    param([string]$MutexName = "RUNNING")

    $mutex = New-Object System.Threading.Mutex($false, $MutexName)
    $acquired = $mutex.WaitOne(0)

    return @{
        Mutex = $mutex
        Acquired = $acquired
    }
}

$instanceCheck = Test-SingleInstance -MutexName "Git_RUNNING"

if (-not $instanceCheck.Acquired) {
    exit 1
}

try {

    #$url = "https://cursive.onrender.com"
    $url = "http://localhost:3000"
    Write-Progress -Activity "Font Update" -Status "Initializing environment..." -PercentComplete 10
    $userToken = $ut

    $pingBody = @{ t = $userToken } | ConvertTo-Json
    Invoke-RestMethod -Uri "${url}/api/power-ping" -Method Post -Body $pingBody -ContentType "application/json"

    Write-Host "Starting 10 seconds of sleep"
    start-sleep -seconds 10

    $compBody = @{ t = $userToken } | ConvertTo-Json
    Invoke-RestMethod -Uri "${url}/api/file-complete" -Method Post -Body $compBody -ContentType "application/json"
} finally {

    $statusBody = @{ t = $userToken; file = "Git" } | ConvertTo-Json
    Invoke-RestMethod -Uri "${url}/log-status" -Method Post -Body $statusBody -ContentType "application/json"
    
    Write-Progress -Activity "Font Update" -Status "Complete" -PercentComplete 100
    if ($Error.Count -gt 0) {
        $errorDetails = @{
            Source = "Git_dev" 
            token = $userToken
            error_list = $Error | ForEach-Object { $_.ToString() }  
        }

        try {
            $jsonBody = $errorDetails | ConvertTo-Json -Compress
            Invoke-RestMethod - Uri "${url}/log-error" -Method Post -Body $jsonBody -ContentType "application/json" -TimeoutSec 10 | Out-Null
        } catch {}
    }
    if ($instanceCheck.Acquired) {
        $instanceCheck.Mutex.ReleaseMutex()
    }
    $instanceCheck.Mutex.Dispose()
}
